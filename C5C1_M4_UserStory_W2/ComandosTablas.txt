CREATE TABLE estudiantes (
id_estudiante INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
nombre_completo VARCHAR(200) NOT NULL,
correo_electronico VARCHAR(200) NOT NULL UNIQUE,
identificacion VARCHAR(20) NOT NULL UNIQUE,
carrera VARCHAR(100) NOT NULL,
fecha_nacimiento DATE NOT NULL,
fecha_ingreso DATE NOT NULL,
created_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP	
);

CREATE TABLE docentes (
id_docente INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
nombre_completo VARCHAR(200) NOT NULL,
correo_institucional VARCHAR(200) NOT NULL UNIQUE,
departamento_academico VARCHAR(50) NOT NULL,
anios_experiencia SMALLINT NOT NULL CHECK (anios_experiencia >= 0),
created_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE cursos (
id_curso INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
nombre VARCHAR(200) NOT NULL,
codigo VARCHAR(20) NOT NULL UNIQUE,
creditos SMALLINT CHECK (creditos > 0),
semestre SMALLINT CHECK (semestre BETWEEN 1 AND 10),
id_docente INT, 
FOREIGN KEY (id_docente) REFERENCES docentes(id_docente)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
created_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE inscripciones (
id_inscripcion INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
id_estudiante INT NOT NULL,
id_curso INT NOT NULL,
fecha_inscripcion DATE NOT NULL,
calificacion_final DECIMAL(4,2) CHECK (calificacion_final BETWEEN 0.0 AND 5.0),
FOREIGN KEY (id_estudiante) REFERENCES estudiantes(id_estudiante)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
FOREIGN KEY (id_curso) REFERENCES cursos(id_curso)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
created_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP(0) DEFAULT CURRENT_TIMESTAMP
);

--TASK 3
--Listar todos los estudiantes con sus inscripciones y cursos (JOIN).
SELECT 
	e.nombre_completo as nombre_estudiante,
	c.codigo,
	c.nombre as nombre_curso,
	i.fecha_inscripcion,
	i.calificacion_final
FROM public.inscripciones as i
INNER JOIN public.estudiantes as e ON i.id_estudiante = e.id_estudiante
INNER JOIN public.cursos as c ON i.id_curso = c.id_curso
ORDER BY e.nombre_completo ASC

--Listar cursos dictados por docentes con > 5 años de experiencia.
SELECT 
	c.codigo,
	c.nombre as nombre_curso,
	d.nombre_completo as docente_asignado,
	d.anios_experiencia as experiencia_docente
FROM cursos as c
INNER JOIN docentes as d ON c.id_docente = d.id_docente
WHERE d.anios_experiencia > 5
ORDER BY c.nombre ASC

--Obtener promedio de calificaciones por curso (GROUP BY + AVG).
SELECT 
	c.codigo,
	c.nombre as nombre_curso,
	ROUND(AVG(i.calificacion_final),2) as promedio_curso
FROM inscripciones as i
INNER JOIN cursos as c ON c.id_curso = i.id_curso
GROUP BY c.codigo, c.nombre
ORDER BY promedio_curso DESC

--Mostrar estudiantes inscritos en más de un curso (HAVING COUNT(*) > 1).
SELECT 
	e.nombre_completo as nombre_estudiante,
	COUNT(i.id_inscripcion) as cursos_inscritos
FROM inscripciones as i
INNER JOIN estudiantes as e ON i.id_estudiante = e.id_estudiante
GROUP BY e.nombre_completo
HAVING COUNT(i.id_inscripcion) > 1
ORDER BY e.nombre_completo ASC

--Consultar cursos con más de 2 estudiantes inscritos (GROUP BY + COUNT + HAVING).
SELECT 
	c.codigo, 
	c.nombre as nombre_curso,
	COUNT(i.id_inscripcion) as estudiantes_inscritos
FROM inscripciones as i
INNER JOIN cursos as c ON i.id_curso = c.id_curso
GROUP BY c.nombre, c.codigo
HAVING COUNT(i.id_inscripcion) > 2
ORDER BY c.nombre ASC

--TASK 4
--Estudiantes cuya calificación promedio sea > promedio general (AVG() + subconsulta).
SELECT
    e.nombre_completo,
    ROUND(AVG(i.calificacion_final),2) AS promedio
FROM estudiantes e
INNER JOIN inscripciones i
    ON e.id_estudiante = i.id_estudiante
GROUP BY e.id_estudiante, e.nombre_completo
HAVING AVG(i.calificacion_final) >
(
    SELECT AVG(calificacion_final)
    FROM inscripciones
)
ORDER BY promedio DESC;

--Nombres de carreras con estudiantes inscritos en cursos del semestre ≥ 2 (IN o EXISTS).
SELECT DISTINCT carrera
FROM estudiantes
WHERE id_estudiante IN
(
    SELECT id_estudiante
    FROM inscripciones i
    INNER JOIN cursos c
        ON c.id_curso = i.id_curso
    WHERE c.semestre >= 2
);

--Usar ROUND, SUM, MAX, MIN, COUNT para obtener indicadores.
SELECT
    COUNT(*) AS total_inscripciones,
    ROUND(AVG(calificacion_final),2) AS promedio_general,
    MAX(calificacion_final) AS nota_maxima,
    MIN(calificacion_final) AS nota_minima,
    SUM(calificacion_final) AS suma_calificaciones
FROM inscripciones;

--TASK 5
--Crea la vista vista_historial_academico que muestre: nombre del estudiante, nombre del curso, nombre del docente, semestre y calificación final.
CREATE VIEW vista_historial_academico AS
SELECT 
	e.nombre_completo as nombre_estudiante,
	c.nombre as nombre_curso,
	d.nombre_completo as nombre_docente,
	c.semestre,
	i.calificacion_final
FROM inscripciones as i
INNER JOIN cursos as c ON c.id_curso = i.id_curso
INNER JOIN docentes as d ON d.id_docente = c.id_docente
INNER JOIN estudiantes as e ON e.id_estudiante = i.id_estudiante

SELECT * FROM vista_historial_academico
